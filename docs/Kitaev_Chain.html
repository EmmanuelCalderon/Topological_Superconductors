# Cadena de Kitaev

lat=[[1.0,0.0],[0.0,1.0]]
orb=[[0.0,-1.0],[0.,1.0]]

t = 1
Δ = 0.5
Nμ=500
KitaevChain=[]
M=np.linspace(-3*t, 3*t, num=Nμ)
for μ in M:
  KC=tb_model(1,2,lat,orb,per=[0])
  KC.set_hop(-t,0,0,[1,0]) # [1] : significa e^{i([1].a1)(k)}
  KC.set_hop(t,1,1,[1,0])
  KC.set_hop(-Δ,0,1,[1,0])
  KC.set_hop( Δ,0,1,[-1,0])
  KC.set_onsite([-μ,μ])
  KitaevChain.append(KC)

KitaevChain[1].visualize(0,1)
